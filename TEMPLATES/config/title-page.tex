\newcommand{\lugtitlepage}{
    \thispagestyle{lugtitle}
    \begin{titlepage}
        \centering
        
        % Logo space (if available)
        \vspace{2cm}
        
        % Main title
        \titlesize\lugbold\color{lugprimary}
        \lugsessiontitle
        
        \vspace{1cm}
        
        % Subtitle
        \subtitlesize\color{lugmedium}
        \lugseries\ \#\lugsessionnumber
        
        \vspace{0.5cm}
        
        % Organization
        \large\color{lugdark}
        \lugname
        
        \vspace{2cm}
        
        % Abstract box
        \begin{center}
            \colorbox{luglight}{
                \begin{minipage}{0.85\textwidth}
                    \centering
                    \vspace{0.5cm}
                    \Large\lugbold{Document Abstract}
                    \vspace{0.3cm}
                    
                    \normalsize
                    \lugsessiondescription
                    \vspace{0.5cm}
                \end{minipage}
            }
        \end{center}
        
        \vfill
        
        % Session info
        \begin{center}
            \begin{tabular}{ll}
                \lugbold{Session:} & \lugsessionnumber \\
                \lugbold{Date:} & \lugsessiondate \\
                \lugbold{Location:} & \luglocation \\
                \lugbold{Website:} & \lugwebsite \\
            \end{tabular}
        \end{center}
        
        \vspace{1cm}
        
        % License
        \begin{center}
            \small\lugitalic{\luglicensetext}
        \end{center}
        
    \end{titlepage}
    \clearpage
}

%----------------------------------------------
% Cover Page with Background
%----------------------------------------------
\newcommand{\lugcoverpage}{
    \thispagestyle{lugtitle}
    \begin{titlepage}
        \begin{tikzpicture}[remember picture,overlay]
            \fill[lugprimary] (current page.south west) rectangle (current page.north east);
            \fill[luglight] ([yshift=-5cm]current page.north west) rectangle (current page.south east);
        \end{tikzpicture}
        
        \vspace{3cm}
        \centering
        
        \color{white}
        \titlesize\lugbold
        \lugsessiontitle
        
        \vspace{1cm}
        
        \color{lugprimary}
        \subtitlesize\lugbold
        \lugseries\ \#\lugsessionnumber
        
        \vspace{0.5cm}
        
        \color{lugdark}
        \large
        \lugname
        
        \vfill
        
        \color{lugdark}
        \normalsize\lugdate{\lugsessiondate}
        
        \vspace{2cm}
        
    \end{titlepage}
    \clearpage
}
