\usepackage{tcolorbox}
\tcbuselibrary{most}

% Info box
\newtcolorbox{infobox}[1][]{
	colback=lugblue!10,
	colframe=lugblue,
	title=Information,
	fonttitle=\bfseries,
	#1
}

% Warning box
\newtcolorbox{warningbox}[1][]{
	colback=orange!10,
	colframe=orange!80,
	title=Important Note,
	fonttitle=\bfseries,
	#1
}

% Command box
\newtcolorbox{commandbox}[1][]{
	colback=luglight,
	colframe=lugdark,
	title=Command Line,
	fonttitle=\bfseries\ttfamily,
	#1
}
